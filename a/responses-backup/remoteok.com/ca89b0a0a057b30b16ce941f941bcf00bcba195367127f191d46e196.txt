					$('.feedback-popup').removeClass('success');
					$('.feedback-popup input').val('');
					$('.feedback-popup textarea').val('');

				}

					$('.feedback-popup input').val('');
					$('.feedback-popup textarea').val('');
					$('.feedback-popup').removeClass('success');
					$('.feedback-popup .fa').addClass('fa-chevron-up').removeClass('fa-chevron-down');

					}
